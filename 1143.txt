using System;

class Program
{
    static void Main()
    {
        int N = int.Parse(Console.ReadLine());

        if (N <= 1 || N >= 1000)
        {
            Console.WriteLine("N deve estar entre 1 e 1000.");
            return;
        }

        for (int i = 1; i <= N; i++)
        {
            int quadrado = i * i;
            int cubo = i * i * i;

            Console.WriteLine($"{i} {quadrado} {cubo}");
        }
    }
}
